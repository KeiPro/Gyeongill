	inta=3(x)    int a = 3(o)

	3) 실행문의 끝을 ;(세미콜론)으로 마무리한다.
	int a = 3;  int b = 5;

	모든 C++ 프로그램은 최소 1개의 main함수를 가진다.
	C++ 프로그램이 실행된다는 것 == main함수안의 내용이 실행된다.
	main함수를 실행시키기 위해서는 소스프로그램 => (컴파일과 링크를 거쳐) => exe파일 생성(exe 실행)
	런타임(실행시간)

*/

	#include <파일명>   <- CPP 라이브러리 파일을 포함시킬 때.
	#include 는 소스파일의 가장 상위에 위치해야 한다.
*/

using namespace std;

// 프로그램 최초 실행 함수
int main()
{	
	// 헬로 월드를 출력하는 코드를 작성

	cout << 10 << endl;

	cout << "너의 전재산은 " << 100 * 2000 << "원이다." <<endl;

	// 기본적인 실행순서 = 위에서 아래로
	// 실행순서를 컨트롤 = 제어구조 (반복문, 조건문)

	// 입력을 하려면 입력받은 데이터를 저장할 공간이 필요하다.(memory)
	// cin : 입력을 받을 때 사용하는 명령어
	// 변수 : 값을 저장할 수 있는 메모리 공간에 붙여진 이름
	// 1) 이름   2) 값   3) 메모리의 주소   4) 크기 (데이터 타입)
	// int age = 15;
	/*
		데이터 타입을 가지는 변수를 '선언'한다.
		메모리 공간이 비어있는 가장 낮은 주소를 찾아서 4바이트 크기의 정수형 데이터를 저장할 공간을 age라는 이름으로 지정한다.
	*/

	int age{10};
/*
	cout << "1. 나이 : " << age << endl;

	cout << "나의 나이를 입력 : ";
	cin >> age;
	
	cout << "나의 나이는 : " << age << endl;
*/

	/*
		변수의 이름 규칙
		1) C++언어에서 사용하는 키워드(keyword)들을 변수 이름으로 사용할 수 없다.
		int int;
		2) 알파벳, 숫자, _로만 구성할 수 있다.
		3) 대소문자 구분 myAge, myage는 다르다.
		4) 변수의 이름을 숫자로 시작할 수 없다. 1num .. (x), _2num .. (o)
		5) 이름 사이에 띄어쓰기를 넣을 수 없다. C++에서는 띄어쓰기로 요소를 구분하기 때문에
		my age(x) my_age(o)

		데이터 타입 = 변수가 가지는 데이터의 형태 '자료형'
		변수의 타입 = 정수형, 실수형, 문자형, 문자열, Boolean형
		
		1) 정수형
		변수 타입명		크기(byte)			범위
		int					4			-21억 ~ +21억
		long long			8			
		short				2			-32768 ~ +32767
		__int8				1
		__int16				2
		__int32				4
		__int64				8

		unsigned 접두어를 붙여서 양수 형태로만 쓸 수 있다.

		2) 실수형 변수(부동소수점 실수)
		float				4
		double				8

		3) 문자형 변수
		char				1	1개의 문자를 표현한다. 한글 ㄱㄴㄷㄹ은 최소 2바이트로 표현.

		4) 문자열 변수
		string				문자열의 총 크기 + 1 (널null 문자)
		
		5) Boolean형 변수
		bool				1	값은 0 / 1 (false / true)

		자료형이나 변수의 크기는 sizeof()연산자를 통해서 측정할 수 있다.
	*/

	string name = "김영우";
	cout << "name변수의 크기는 " << name.size() << endl;
	cout << "__int64형의 크기는 " << sizeof(__int64) << endl;

	//실습1.
	//float형 변수를 하나 선언해서 초기값을 세팅하고
	//sizeof연산자를 통해 값과 크기를 출력하라.
	float k = 10.512f; //리터럴
	cout << "해당 값은 " << k << "이고 변수의 크기는 " << sizeof(k) << "입니다." <<endl;

	*/
